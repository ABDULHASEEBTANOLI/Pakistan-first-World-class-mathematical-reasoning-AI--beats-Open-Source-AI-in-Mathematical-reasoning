// MatBERT — Entity Extraction (Training Time) 
from transformers import AutoTokenizer, AutoModelForTokenClassification
import torch

# MatBERT — materials science specific NER
matbert = AutoModelForTokenClassification.from_pretrained("m3rg-iitd/matscibert")
tokenizer = AutoTokenizer.from_pretrained("m3rg-iitd/matscibert")

def extract_entities_matbert(paper_text):
    """
    Paper se yeh cheezein extract karta hai:
    - Material names: Fe3C, Ti-6Al-4V, Ni-based superalloy
    - Properties: hardness, melting point, Young's modulus
    - Numerical values: 800 HV, 1538°C, 210 GPa
    - Processing conditions: annealing, quenching, sintering
    """
    inputs = tokenizer(paper_text, return_tensors="pt", truncation=True, max_length=512)
    outputs = matbert(**inputs)
    entities = parse_ner_output(outputs, inputs)
    return {
        "materials": entities["MAT"],
        "properties": entities["PRO"],
        "values": entities["VAL"],
        "processes": entities["PRO_COND"],
    }

def matbert_embed(text):
    # ChromaDB ke liye superior domain embeddings
    inputs = tokenizer(text, return_tensors="pt", truncation=True, max_length=512)
    with torch.no_grad():
        outputs = matbert.bert(**inputs)
    return outputs.last_hidden_state[:, 0, :].numpy()[0].tolist()

def route_query(user_query, llm):
    result = llm.quick_classify(ROUTER_PROMPT.format(user_query=user_query))
    routing = json.loads(result)

    if routing["category"] == "FACTUAL_CURRENT" or routing["confidence"] < 7:
        web_results = tavily_search(user_query)

// WEB SEARCH SAFETY — Internet results sanitizer 
def sanitize_web_results(raw_results: list) -> str:
    """
    Web se ane wali content ko filter karo:
    - Irrelevant results hatao (cosine similarity < 0.5 with query)
    - Inject ONLY as read-only context string
    - Model weights kabhi nahi badte — yeh sirf input text hai
    - Max 3 results, max 300 words each
    """
    filtered = [r for r in raw_results if r["relevance"] > 0.5]
    sanitized = []
    for r in filtered[:3]:
        sanitized.append(
            f"[WEB SOURCE: {r['url']} | Date: {r['date']}]\n{r['snippet'][:300]}"
        )
    return "\n\n".join(sanitized)

import re

def validate_citations(model_output: str, chroma_collection) -> str:
    """
    Model ki output mein har [Source: ...] claim ko ChromaDB mein verify karo.
    Agar source wahan nahi mila → [⚠️ UNVERIFIED] tag lagao.
    """
    citation_pattern = re.compile(r'\[Source:\s*(.+?)\]')
    citations = citation_pattern.findall(model_output)

    for citation in citations:
        results = chroma_collection.query(
            query_texts=[citation],
            n_results=1
        )
        best_distance = results['distances'][0][0] if results['distances'][0] else 1.0

        if best_distance > 0.45:  # threshold — not found
            model_output = model_output.replace(
                f'[Source: {citation}]',
                f'[Source: {citation} | ⚠️ UNVERIFIED — not in knowledge base]'
            )
        else:
            # Add actual metadata from ChromaDB
            meta = results['metadatas'][0][0]
            verified_tag = f'[Source: {meta["title"]} | {meta["year"]} | ✅ VERIFIED]'
            model_output = model_output.replace(f'[Source: {citation}]', verified_tag)

    return model_output

Hybrid Search — BM25 + Semantic NEW
صرف semantic search کافی نہیں — hybrid بہتر ہے
✗ Semantic Only (ChromaDB default)

  • → Exact material codes miss ہو جاتے ہیں

  • → ASTM standards، DOIs exact match نہیں

// HYBRID SEARCH — BM25 + ChromaDB + RRF merge 
from rank_bm25 import BM25Okapi
import numpy as np

class HybridSearchEngine:
    def __init__(self, chroma_collection, documents):
        self.chroma = chroma_collection
        # BM25 index build karo
        tokenized = [doc.lower().split() for doc in documents]
        self.bm25 = BM25Okapi(tokenized)
        self.documents = documents

    def search(self, query: str, top_k: int = 5) -> list:
        # BM25 search — keyword matching
        bm25_scores = self.bm25.get_scores(query.lower().split())
        bm25_top = np.argsort(bm25_scores)[::-1][:top_k * 2]

        # Semantic search — ChromaDB
        semantic_results = self.chroma.query(
            query_texts=[query],
            n_results=top_k * 2
        )
        semantic_ids = semantic_results["ids"][0]

        # Reciprocal Rank Fusion (RRF) — merge rankings
        rrf_scores = {}
        k = 60  # RRF constant
        for rank, doc_id in enumerate(bm25_top):
            rrf_scores[doc_id] = rrf_scores.get(doc_id, 0) + 1 / (k + rank)
        for rank, doc_id in enumerate(semantic_ids):
            rrf_scores[doc_id] = rrf_scores.get(doc_id, 0) + 1 / (k + rank)

        # Sort by RRF score and return top_k
        final = sorted(rrf_scores.items(), key=lambda x: x[1], reverse=True)
        return [self.documents[i] for i, _ in final[:top_k]]

    def multi_query_search(self, queries: list, top_k: int = 5) -> str:
        # Multiple queries → merge results → deduplicate
        all_results = []
        for q in queries:
            all_results.extend(self.search(q, top_k=3))
        unique = list(dict.fromkeys(all_results))  # deduplicate, preserve order
        return "\n\n".join(unique[:top_k])
